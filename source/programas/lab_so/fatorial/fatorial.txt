/* programa para calcular o fatorial de um numero */

int num;
int result;

int factorial(int n) {
    int res;
    int i;

    res = 1;
    i = 1;
    
    while (i <= n) {
        res = res * i;
        i = i + 1;
    }

    return res;
}

void main(void) {
  
    num = input();
    result = factorial(num);
    output(result);
    
}
